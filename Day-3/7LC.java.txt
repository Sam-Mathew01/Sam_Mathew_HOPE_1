class Solution {
    public int reverse(int x) {
        int dig;
        long ans=0;
        int num;
        while(x!=0){
            dig=x%10;
            ans=ans*10+dig;
            x=x/10;
        }
        if(ans>=Integer.MIN_VALUE && ans<=Integer.MAX_VALUE)
            return (int)ans;
        else
            return 0;
    }
}
