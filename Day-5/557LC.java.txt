class Solution {
    public String reverseWords(String s) {
        StringBuilder sb2=new StringBuilder();
        String[] sa=s.split(" ");
        int len=sa.length;
        for(int i=len-1;i>=0;i--){

            sb2.append(sa[i]+" ");

        }
        sb2.reverse();
        return  sb2.toString().trim();
        }
    }
